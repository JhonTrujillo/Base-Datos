//Crear base de datos cerveza
CREATE DATABASE Cerveza;

//Consultar las bases de datos creadas
SHOW DATABASES;

//Usar base de datos creada
USE Cerveza;

//Crear una tabla Cerveza
CREATE TABLE cerveza(
idcerveza INT PRIMARY KEY AUTO_INCREMENT,
nombre varchar(255) NOT NULL,
detalle varchar(255) NOT NULL
);

//Crear una tabla Categoria
CREATE TABLE categoria(
idcategoria INT PRIMARY KEY AUTO_INCREMENT,
nombre varchar(255) NOT NULL,
detalle varchar(255) NOT NULL
);

//Crear una tabla Tipo
CREATE TABLE tipo(
idtipo INT PRIMARY KEY AUTO_INCREMENT,
nombre varchar(255) NOT NULL
);
	

//Crear una tabla MateriaPrima
CREATE TABLE MateriaPrima(
idmateriaprima INT PRIMARY KEY AUTO_INCREMENT,
nombre varchar(255) NOT NULL,
detalle varchar(255) NOT NULL
);

//Agregar llave foranea idcategoria a Tabla cerveza
ALTER TABLE cerveza ADD COLUMN idcategoria INT;
ALTER TABLE cerveza ADD FOREIGN KEY (idcategoria) REFERENCES categoria(idcategoria);

//Crear tabla intermedia tipo_categoria
CREATE TABLE tipo_categoria(
idtipo_categoria INT PRIMARY KEY AUTO_INCREMENT
);

//Agregar llaves foraneas idtipo, idcategoria a tabla tipo_categoria
ALTER TABLE tipo_categoria ADD COLUMN idtipo INT;
ALTER TABLE tipo_categoria ADD FOREIGN KEY (idtipo) REFERENCES tipo(idtipo);

ALTER TABLE tipo_categoria ADD COLUMN idcategoria INT;
ALTER TABLE tipo_categoria ADD FOREIGN KEY (idcategoria) REFERENCES categoria(idcategoria);

//Crear tabla intermedia cerveza_materiaprima(
CREATE TABLE cerveza_materiaprima(
idcerveza_materiaprima INT PRIMARY KEY AUTO_INCREMENT,
cantidad varchar(255) NOT NULL,
unidad varchar(255) NOT NULL
);


//Agregar llaves foraneas idcerveza, idmateriaprima a tabla cerveza_materiaprima
ALTER TABLE cerveza_materiaprima ADD COLUMN idcerveza INT;
ALTER TABLE cerveza_materiaprima ADD FOREIGN KEY (idcerveza) REFERENCES cerveza(idcerveza);

ALTER TABLE cerveza_materiaprima ADD COLUMN idmateriaprima INT;
ALTER TABLE cerveza_materiaprima ADD FOREIGN KEY (idmateriaprima) REFERENCES MateriaPrima(idmateriaprima);
